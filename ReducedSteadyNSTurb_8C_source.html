40ReducedSteadyNSTurb::ReducedSteadyNSTurb(SteadyNSTurb& fomProblem)
41 :
42 problem(&fomProblem)
43{
44 N_BC = problem->inletIndex.rows();
45 Nphi_u = problem->B_matrix.rows();
46 Nphi_p = problem->K_matrix.cols();
47 nphiNut = problem->cTotalTensor.dimension(1);
48
49 for (int k = 0; k < problem->liftfield.size(); k++)
50 {
51 Umodes.append((problem->liftfield[k]).clone());
52 }
53
54 for (int k = 0; k < problem->NUmodes; k++)
55 {
56 Umodes.append((problem->Umodes[k]).clone());
57 }
58
59 for (int k = 0; k < problem->NSUPmodes; k++)
60 {
61 Umodes.append((problem->supmodes[k]).clone());
62 }
63
64 newtonObject = newtonSteadyNSTurb(Nphi_u + Nphi_p, Nphi_u + Nphi_p,
65 fomProblem);
66}

68int newtonSteadyNSTurb::operator()(const Eigen::VectorXd& x,
69 Eigen::VectorXd& fvec) const
70{
71 Eigen::VectorXd aTmp(Nphi_u);
72 Eigen::VectorXd bTmp(Nphi_p);
73 aTmp = x.head(Nphi_u);
74 bTmp = x.tail(Nphi_p);
75 // Convective term
76 Eigen::MatrixXd cc(1, 1);
77 // Mom Term
78 Eigen::VectorXd m1 = problem->bTotalMatrix * aTmp * nu;
79 // Gradient of pressure
80 Eigen::VectorXd m2 = problem->K_matrix * bTmp;
81 // Pressure Term
82 Eigen::VectorXd m3 = problem->P_matrix * aTmp;
83 // Penalty term
84 Eigen::MatrixXd penaltyU = Eigen::MatrixXd::Zero(Nphi_u, N_BC);
85
86 // Term for penalty method
87 if (problem->bcMethod == "penalty")
88 {
89 for (int l = 0; l < N_BC; l++)
90 {
91 penaltyU.col(l) = bc(l) * problem->bcVelVec[l] - problem->bcVelMat[l] *
92 aTmp;
93 }
94 }
95
96 for (int i = 0; i < Nphi_u; i++)
97 {
98 cc = aTmp.transpose() * Eigen::SliceFromTensor(problem->C_tensor, 0,
99 i) * aTmp - gNut.transpose() *
100 Eigen::SliceFromTensor(problem->cTotalTensor, 0, i) * aTmp;
101 fvec(i) = m1(i) - cc(0, 0) - m2(i);
102
103 if (problem->bcMethod == "penalty")
104 {
105 fvec(i) += ((penaltyU * tauU)(i, 0));
106 }
107 }
108
109 for (int j = 0; j < Nphi_p; j++)
110 {
111 int k = j + Nphi_u;
112 fvec(k) = m3(j);
113 }
114
115 if (problem->bcMethod == "lift")
116 {
117 for (int j = 0; j < N_BC; j++)
118 {
119 fvec(j) = x(j) - bc(j);
120 }
121 }
122
123 return 0;
124}

138void ReducedSteadyNSTurb::solveOnlineSUP(Eigen::MatrixXd vel)
139{
140 if (problem->bcMethod == "lift")
141 {
142 vel_now = setOnlineVelocity(vel);
143 }
144 else if (problem->bcMethod == "penalty")
145 {
146 vel_now = vel;
147 }
148
149 y.resize(Nphi_u + Nphi_p, 1);
150 y.setZero();
151
152 // Change initial condition for the lifting function
153 if (problem->bcMethod == "lift")
154 {
155 for (int j = 0; j < N_BC; j++)
156 {
157 y(j) = vel_now(j, 0);
158 }
159 }
160
161 Color::Modifier red(Color::FG_RED);
162 Color::Modifier green(Color::FG_GREEN);
163 Color::Modifier def(Color::FG_DEFAULT);
164 Eigen::HybridNonLinearSolver<newtonSteadyNSTurb> hnls(newtonObject);
165 newtonObject.bc.resize(N_BC);
166 newtonObject.tauU = tauU;
167
168 for (int j = 0; j < N_BC; j++)
169 {
170 newtonObject.bc(j) = vel_now(j, 0);
171 }
172
173 if (problem->viscCoeff == "L2")
174 {
175 for (int i = 0; i < nphiNut; i++)
176 {
177 newtonObject.gNut = problem->nutCoeff;
178 }
179 }
180 else if (problem->viscCoeff == "RBF")
181 {
182 for (int i = 0; i < nphiNut; i++)
183 {
184 newtonObject.gNut(i) = problem->rbfSplines[i]->eval(vel_now);
185 rbfCoeff = newtonObject.gNut;
186 }
187 }
188 else
189 {
190 Info << "The way to compute the eddy viscosity coefficients has to be either L2 or RBF"
191 << endl;
192 exit(0);
193 }
194
195 newtonObject.nu = nu;
196 hnls.solve(y);
197 Eigen::VectorXd res(y);
198 newtonObject.operator()(y, res);
199 std::cout << "################## Online solve N° " << count_online_solve <<
200 " ##################" << std::endl;
201 std::cout << "Solving for the parameter: " << vel_now << std::endl;
202
203 if (res.norm() < 1e-5)
204 {
205 std::cout << green << "|F(x)| = " << res.norm() << " - Minimun reached in " <<
206 hnls.iter << " iterations " << def << std::endl << std::endl;
207 }
208 else
209 {
210 std::cout << red << "|F(x)| = " << res.norm() << " - Minimun reached in " <<
211 hnls.iter << " iterations " << def << std::endl << std::endl;
212 }
213
214 count_online_solve += 1;
215}

218void ReducedSteadyNSTurb::reconstruct(bool exportFields, fileName folder,
219 int printevery)
220{
221 if (exportFields)
222 {
223 mkDir(folder);
224 ITHACAutilities::createSymLink(folder);
225 }
226
227 int counter = 0;
228 int nextWrite = 0;
229 List <Eigen::MatrixXd> CoeffU;
230 List <Eigen::MatrixXd> CoeffP;
231 List <Eigen::MatrixXd> CoeffNut;
232 CoeffU.resize(0);
233 CoeffP.resize(0);
234 CoeffNut.resize(0);
235
236 for (int i = 0; i < online_solution.size(); i++)
237 {
238 if (counter == nextWrite)
239 {
240 Eigen::MatrixXd currentUCoeff;
241 Eigen::MatrixXd currentPCoeff;
242 Eigen::MatrixXd currentNutCoeff;
243 currentUCoeff = online_solution[i].block(1, 0, Nphi_u, 1);
244 currentPCoeff = online_solution[i].bottomRows(Nphi_p);
245 currentNutCoeff = rbfCoeffMat.block(0, i, nphiNut, 1);
246 CoeffU.append(currentUCoeff);
247 CoeffP.append(currentPCoeff);
248 CoeffNut.append(currentNutCoeff);
249 nextWrite += printevery;
250 }
251
252 counter++;
253 }
254
255 volVectorField uRec("uRec", Umodes[0]);
256 volScalarField pRec("pRec", problem->Pmodes[0]);
257 volScalarField nutRec("nutRec", problem->nutModes[0]);
258 uRecFields = problem->L_U_SUPmodes.reconstruct(uRec, CoeffU, "uRec");
259 pRecFields = problem->Pmodes.reconstruct(pRec, CoeffP, "pRec");
260 nutRecFields = problem->nutModes.reconstruct(nutRec, CoeffNut, "nutRec");
261
262 if (exportFields)
263 {
264 ITHACAstream::exportFields(uRecFields, folder,
265 "uRec");
266 ITHACAstream::exportFields(pRecFields, folder,
267 "pRec");
268 ITHACAstream::exportFields(nutRecFields, folder,
269 "nutRec");
270 }
271}

273Eigen::MatrixXd ReducedSteadyNSTurb::setOnlineVelocity(Eigen::MatrixXd vel)
274{
275 assert(problem->inletIndex.rows() == vel.rows()
276 && "Imposed boundary conditions dimensions do not match given values matrix dimensions");
277 Eigen::MatrixXd vel_scal;
278 vel_scal.resize(vel.rows(), vel.cols());
279
280 for (int k = 0; k < problem->inletIndex.rows(); k++)
281 {
282 int p = problem->inletIndex(k, 0);
283 int l = problem->inletIndex(k, 1);
284 scalar area = gSum(problem->liftfield[0].mesh().magSf().boundaryField()[p]);
285 scalar u_lf = gSum(problem->liftfield[k].mesh().magSf().boundaryField()[p] *
286 problem->liftfield[k].boundaryField()[p]).component(l) / area;
287 vel_scal(k, 0) = vel(k, 0) / u_lf;
288 }
289
290 return vel_scal;
291}
