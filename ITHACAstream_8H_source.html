550template <typename T, int whatever, typename IND>
551void ITHACAstream::SaveSparseMatrix(Eigen::SparseMatrix<T, whatever, IND>& m,
552 word folder, word MatrixName)
553{
554 typedef Eigen::Triplet<int> Trip;
555 std::vector<Trip> res;
556 m.makeCompressed();
557 mkDir(folder);
558 std::fstream writeFile;
559 writeFile.open(folder + MatrixName, std::ios::binary | std::ios::out);
560
561 if (writeFile.is_open())
562 {
563 IND rows, cols, nnzs, outS, innS;
564 rows = m.rows() ;
565 cols = m.cols() ;
566 nnzs = m.nonZeros() ;
567 outS = m.outerSize();
568 innS = m.innerSize();
569 writeFile.write(reinterpret_cast<const char*> (&rows), sizeof(IND));
570 writeFile.write(reinterpret_cast<const char*> (&cols), sizeof(IND));
571 writeFile.write(reinterpret_cast<const char*> (&nnzs), sizeof(IND));
572 writeFile.write(reinterpret_cast<const char*> (&outS), sizeof(IND));
573 writeFile.write(reinterpret_cast<const char*> (&innS), sizeof(IND));
574 writeFile.write(reinterpret_cast<const char*>(m.valuePtr()),
575 sizeof(T ) * m.nonZeros());
576 writeFile.write(reinterpret_cast<const char*>(m.outerIndexPtr()),
577 sizeof(IND) * m.outerSize());
578 writeFile.write(reinterpret_cast<const char*>(m.innerIndexPtr()),
579 sizeof(IND) * m.nonZeros());
580 writeFile.close();
581 }

584template <typename T, int whatever, typename IND>
585void ITHACAstream::ReadSparseMatrix(Eigen::SparseMatrix<T, whatever, IND>& m,
586 word folder, word MatrixName)
587{
588 std::fstream readFile;
589 readFile.open(folder + MatrixName, std::ios::binary | std::ios::in);
590
591 if (!readFile.good())
592 {
593 std::cout << folder + MatrixName <<
594 " file does not exist, try to rerun the Offline Stage!" << std::endl;
595 exit(EXIT_FAILURE);
596 }
597 else if (readFile.is_open())
598 {
599 IND rows, cols, nnz, inSz, outSz;
600 readFile.read(reinterpret_cast<char*>(&rows ), sizeof(IND));
601 readFile.read(reinterpret_cast<char*>(&cols ), sizeof(IND));
602 readFile.read(reinterpret_cast<char*>(&nnz ), sizeof(IND));
603 readFile.read(reinterpret_cast<char*>(&inSz ), sizeof(IND));
604 readFile.read(reinterpret_cast<char*>(&outSz), sizeof(IND));
605 m.resize(rows, cols);
606 m.makeCompressed();
607 m.resizeNonZeros(nnz);
608 readFile.read(reinterpret_cast<char*>(m.valuePtr()), sizeof(T ) * nnz );
609 readFile.read(reinterpret_cast<char*>(m.outerIndexPtr()), sizeof(IND) * outSz);
610 readFile.read(reinterpret_cast<char*>(m.innerIndexPtr()), sizeof(IND) * nnz );
611 m.finalize();
612 readFile.close();
613 }

616template <typename MatrixType>
617void ITHACAstream::SaveDenseMatrix(MatrixType& Matrix, word folder,
618 word MatrixName)
619{
620 mkDir(folder);
621 std::ofstream out(folder + MatrixName,
622 std::ios::out | std::ios::binary | std::ios::trunc);
623 typename MatrixType::Index rows = Matrix.rows(), cols = Matrix.cols();
624 out.write(reinterpret_cast<char*> (&rows), sizeof(typename MatrixType::Index));
625 out.write(reinterpret_cast<char*> (&cols), sizeof(typename MatrixType::Index));
626 out.write(reinterpret_cast<char*> (Matrix.data()),
627 rows * cols * sizeof(typename MatrixType::Scalar) );
628 out.close();
629}

657template <typename TensorType>
658void ITHACAstream::SaveDenseTensor(TensorType& Tensor, word folder,
659 word MatrixName)
660{
661 std::ofstream out(folder + MatrixName,
662 std::ios::out | std::ios::binary | std::ios::trunc);
663 typename TensorType::Dimensions dim = Tensor.dimensions();
664 int tot = 1;
665
666 for (int k = 0; k < dim.size(); k++)
667 {
668 tot *= dim[k];
669 }
670
671 out.write(reinterpret_cast<char*> (&dim),
672 sizeof(typename TensorType::Dimensions));
673 out.write(reinterpret_cast<char*> (Tensor.data()),
674 tot * sizeof(typename TensorType::Scalar) );
675 out.close();
676}

678template <typename TensorType>
679void ITHACAstream::ReadDenseTensor(TensorType& Tensor, word folder,
680 word MatrixName)
681{
682 std::ifstream in;
683 in.open(folder + MatrixName, std::ios::in | std::ios::binary);
684 typename TensorType::Dimensions dim;
685 in.read(reinterpret_cast<char*> (&dim),
686 sizeof(typename TensorType::Dimensions));
687 auto dims = Tensor.dimensions();
688 M_Assert(dims.size() == dim.size(),
689 "The rank of the tensor you want to fill does not coincide with the rank of the tensor you are reading");
690 int tot = 1;
691
692 for (int k = 0; k < dim.size(); k++)
693 {
694 tot *= dim[k];
695 }
696
697 Tensor.resize(dim);
698 in.read( reinterpret_cast<char*>(Tensor.data()),
699 tot * sizeof(typename TensorType::Scalar) );
700 in.close();
701}
